keep i at initial position. keep j at i+1. Store the difference
TC: O(N^2)
SC: O(1)


class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n=prices.size();
        int maxPro=0;
        int minPrice=INT_MAX;
        for(int i=0;i<n;i++){
            minPrice=min(minPrice,prices[i]);
            maxPro=max(maxPro,prices[i]-minPrice);
        }
        return maxPro;
    }
};

TC: O(N)
SC: O(1)
